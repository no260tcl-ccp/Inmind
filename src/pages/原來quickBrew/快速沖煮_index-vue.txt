<template>
  <div class="page-container">
    
    <div class="navbar">
      <div class="nav-title">快速沖煮 (除錯模式)</div>
      <div class="device-status">
        <span class="status-dot" :class="{ active: blueStore.connectedDevice }"></span>
        {{ blueStore.connectedDevice?.name || '未連接' }}
      </div>
    </div>

    <div class="control-panel">
      <div class="control-item">
        <div class="label">粉水比 (1:{{ ratio }})</div>
        <input type="range" min="10" max="20" v-model="ratio" class="slider" />
      </div>
      <div class="control-item">
        <div class="label">水溫 ({{ temperature }}°C)</div>
        <input type="range" min="80" max="100" v-model="temperature" class="slider" />
      </div>
      <div class="control-item">
        <div class="label">研磨度 ({{ grindLevel }})</div>
        <div class="stepper">
          <button @click="grindLevel > 1 && grindLevel--">-</button>
          <span>{{ grindLevel }}</span>
          <button @click="grindLevel < 10 && grindLevel++">+</button>
        </div>
      </div>
      <div class="control-item">
        <div class="label">豆粉量 ({{ beanAmount }}g)</div>
        <input type="range" min="10" max="30" v-model="beanAmount" class="slider" />
      </div>
    </div>

    <div class="action-area">
      <button class="btn-start" @click="startBrew" :disabled="isBusy">
        {{ isBusy ? '執行中...' : '☕ 開始沖煮流程' }}
      </button>
      <button class="btn-back" @click="goBack">返回首頁</button>
    </div>

    <div class="debug-console">
      <div class="console-header">
        <span>執行日誌</span>
        <button class="btn-clear" @click="logs = []">清空</button>
      </div>
      <div class="console-body" ref="logBox">
        <div v-if="logs.length === 0" class="log-empty">暫無日誌...</div>
        <div 
          v-for="(log, index) in logs" 
          :key="index" 
          class="log-item"
          :class="log.type"
        >
          <span class="log-time">[{{ log.time }}]</span>
          <span class="log-tag">{{ log.tag }}</span>
          <span class="log-msg">{{ log.msg }}</span>
        </div>
      </div>
    </div>

  </div>
</template>

<script setup>
/* eslint-disable */
import { ref, watch, nextTick, onMounted } from 'vue'
import { useBluetoothStore } from '@/store/blue.js'

const blueStore = useBluetoothStore()
const logs = ref([])
const logBox = ref(null)
const isBusy = ref(false)

// 頁面參數
const ratio = ref(15)
const temperature = ref(92)
const grindLevel = ref(5)
const beanAmount = ref(15)

// === 日誌輔助工具 ===
const addLog = (tag, msg, type = 'info') => {
  const time = new Date().toLocaleTimeString('en-US', { hour12: false, hour: "numeric", minute: "numeric", second: "numeric" })
  logs.value.push({ time, tag, msg, type })
  // 自動捲動到底部
  nextTick(() => {
    if (logBox.value) logBox.value.scrollTop = logBox.value.scrollHeight
  })
}

// === Buffer 轉 Hex 字串 (方便閱讀) ===
const buf2hex = (buffer) => {
  return [...new Uint8Array(buffer)]
    .map(x => x.toString(16).padStart(2, '0').toUpperCase())
    .join(' ')
}

// === 監聽機器回傳數據 ===
// 當 Store 收到數據時，會更新 receivedData，我們這裡監聽它
watch(() => blueStore.receivedData, (newVal) => {
  if (newVal && newVal.data) {
    const hex = buf2hex(newVal.data)
    addLog('RX', `收到: ${hex}`, 'success')
  }
})

// === 輔助函式 ===
const sleep = (ms) => new Promise(resolve => setTimeout(resolve, ms))

const calcChecksum = (view, dataLen) => {
  let checksum = view.getUint8(1) ^ view.getUint8(2) // CMD ^ Len
  for (let i = 0; i < dataLen; i++) {
    checksum ^= view.getUint8(3 + i)
  }
  return checksum
}

// ==========================================
// 核心流程: 喚醒 -> 寫配方 -> 啟動
// ==========================================
const startBrew = async () => {
  if (isBusy.value) return
  isBusy.value = true
  logs.value = [] // 清空舊日誌
  addLog('SYS', '開始執行沖煮流程...')

  try {
    // ------------------------------------------------
    // 1. 喚醒 (Wake Up) CMD: 0x00
    // ------------------------------------------------
    addLog('STEP 1', '發送喚醒指令 (0x00)')
    const wakeUpBuf = new ArrayBuffer(6)
    const wakeView = new DataView(wakeUpBuf)
    wakeView.setUint8(0, 0x5A); wakeView.setUint8(1, 0x00); wakeView.setUint8(2, 0x01);
    wakeView.setUint8(3, 0x00); // 00: 退出待機
    wakeView.setUint8(4, 0x01); // Checksum
    wakeView.setUint8(5, 0xAA);

    addLog('TX', buf2hex(wakeUpBuf), 'tx')
    await blueStore.sendData(wakeUpBuf)
    await sleep(500)

    // ------------------------------------------------
    // 2. 寫入配方 (Set Recipe) CMD: 0x06
    // ------------------------------------------------
    addLog('STEP 2', '寫入配方參數 (0x06)')
    // 根據 PDF: Head(1)+CMD(1)+Len(1)+Data(55)+Check(1)+Tail(1) = 60
    const recipeBuf = new ArrayBuffer(60)
    const rView = new DataView(recipeBuf)
    
    rView.setUint8(0, 0x5A)
    rView.setUint8(1, 0x06)
    rView.setUint8(2, 55) // Len

    // Data 0-3: ID (00 00 00 01)
    rView.setUint8(3,0); rView.setUint8(4,0); rView.setUint8(5,0); rView.setUint8(6,1);

    // Data 4-8: 基本參數
    rView.setUint8(7, parseInt(ratio.value))      
    rView.setUint8(8, parseInt(beanAmount.value)) 
    rView.setUint8(9, 0x01) // 研磨開
    rView.setUint8(10, parseInt(grindLevel.value))
    rView.setUint8(11, 20)  // 轉速

    // Data 9: 段數 (1段)
    rView.setUint8(12, 0x01)

    // Data 10-14: 第0段參數
    rView.setUint8(13, 50)               // 水量
    rView.setUint8(14, 20)               // 流速
    rView.setUint8(15, parseInt(temperature.value)) // 溫度
    rView.setUint8(16, 0x00)             // 方式
    rView.setUint8(17, 0x00)             // 停留

    // 填充剩餘 0
    for(let i=18; i<38; i++) rView.setUint8(i, 0)

    // Data 35-54: 名稱
    const nameStr = "APP-TEST"
    for(let i=0; i<20; i++) {
        rView.setUint8(38+i, i < nameStr.length ? nameStr.charCodeAt(i) : 0)
    }

    // Checksum & Tail
    rView.setUint8(58, calcChecksum(rView, 55))
    rView.setUint8(59, 0xAA)

    addLog('TX', buf2hex(recipeBuf), 'tx')
    await blueStore.sendData(recipeBuf)
    await sleep(800) // 寫入配方需要較長時間

    // ------------------------------------------------
    // 3. 啟動 (Start) CMD: 0x04
    // ------------------------------------------------
    addLog('STEP 3', '發送啟動指令 (0x04)')
    const startBuf = new ArrayBuffer(15)
    const sView = new DataView(startBuf)
    
    sView.setUint8(0, 0x5A)
    sView.setUint8(1, 0x04)
    sView.setUint8(2, 0x0A) // Len 10

    sView.setUint8(3, 0x01) // Start
    sView.setUint8(4, 0x00)
    // ID 1
    sView.setUint8(5,0); sView.setUint8(6,0); sView.setUint8(7,0); sView.setUint8(8,1);
    sView.setUint8(9, 0x01) // 段數 1
    sView.setUint8(10,0); sView.setUint8(11,0); sView.setUint8(12,0);

    sView.setUint8(13, calcChecksum(sView, 10))
    sView.setUint8(14, 0xAA)

    addLog('TX', buf2hex(startBuf), 'tx')
    await blueStore.sendData(startBuf)
    
    addLog('SYS', '指令發送完畢，等待機器反應...', 'success')

  } catch (error) {
    addLog('ERR', error.message || error, 'error')
    console.error(error)
  } finally {
    isBusy.value = false
  }
}

const goBack = () => {
  if (blueStore.connectedDevice) blueStore.disconnectFromDevice(blueStore.connectedDevice.deviceId)
  uni.navigateTo({ url: '/pages/index/index' })
}
</script>
