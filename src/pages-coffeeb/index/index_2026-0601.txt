<script setup lang="ts">
// src\pages-coffeeb\index\index.vue
import { ref, computed, onMounted } from 'vue'
import { useRouter } from 'vue-router'
import { showToast, showConfirmDialog } from 'vant'
import { useMachineBStatusStore } from '@/store/coffeebStatus'
import { useBluetoothStore } from '@/store/blue'

// ✅ 引入您的 Web BLE Adapter，接管所有底層藍牙通訊
//import uni from '@/utils/web-ble-adapter'
//import * as uni from '@/utils/web-ble-adapter'
// ✅ 將所有功能打包成 uniAdapter 變數 (請保留您能成功編譯的路徑)
import * as uniAdapter from '@/utils/web-ble-adapter'

const router = useRouter()
const bluetoothStore = useBluetoothStore()
const machineStatusStore = useMachineBStatusStore()

const settingShow = ref(false)
const colorValue = ref('white')
const homeImg = ref('https://cdn.bincoocoffee.cn/home-machine.png')

const recipeShow = ref(false)
const currentTab = ref('local')

const showSaveDialog = ref(false)
const newRecipeName = ref('')

// 模擬當前機器/首頁設定的參數 (後續可從 store 取得)
const currentPowder = ref(15)
const currentRatio = ref(15)
const currentTemp = ref(92)

interface Recipe {
  id: number | string;
  name: string;
  powder: number;
  ratio: number;
  temp: number;
}

const localRecipes = ref<Recipe[]>([])
const cloudRecipes = ref<Recipe[]>([
  { id: 101, name: '【官方】世界冠军手冲参数', powder: 15, ratio: 15, temp: 92 },
  { id: 102, name: '【官方】深焙豆黄金萃取', powder: 18, ratio: 12, temp: 88 }
])

const openSaveDialog = () => {
  newRecipeName.value = '' 
  showSaveDialog.value = true
}

const deleteLocalRecipe = (index: number) => {
  showConfirmDialog({
    title: '删除配方',
    message: '确定要删除这个私藏配方吗？',
  }).then(() => {
    localRecipes.value.splice(index, 1)
    localStorage.setItem('bincoo_local_recipes', JSON.stringify(localRecipes.value))
    showToast({ type: 'success', message: '已删除' })
  }).catch(() => {})
}

// ✅ 透過 Web BLE Adapter 發送 0x06 配方指令寫入咖啡機
// ✅ 透過 Web BLE Adapter 發送 0x06 配方指令 (終極防呆版)

const shareRecipeToDevice = (recipe: Recipe) => {
  if (!bluetoothStore.connectedDevice) {
    showToast({ type: 'fail', message: '分享失败：请先连接咖啡机蓝牙' })
    return
  }
  
  try {
    // 1. 建立 55 Bytes 陣列
    const recipePkg = new Uint8Array(55)
    
    recipePkg[0] = 0x5A  // 起始符
    recipePkg[1] = 0x06  // CMD: 配方数据
    recipePkg[2] = 49    // 数据长度
    
    // Data 段填充
    recipePkg[3] = 0x01          // 配方 ID
    recipePkg[4] = recipe.powder // 豆粉量
    recipePkg[5] = recipe.ratio  // 粉水比
    recipePkg[6] = recipe.temp   // 水温
    
    for (let i = 7; i <= 51; i++) {
      recipePkg[i] = 0x00 // 其余注水阶段补0
    }

    // 计算 Crc (从 CMD 到 数据段结束进行异或)
    let crc = recipePkg[1]
    for (let i = 2; i <= 51; i++) {
      crc ^= recipePkg[i]
    }
    recipePkg[52] = crc  
    recipePkg[53] = 0xAA 

    console.log("🔥 準備發送 0x06 專屬配方:", recipePkg)

    // ==========================================
    // 2. 🤖 自動偵測：尋找真實的藍牙發送函數
    // ==========================================
    let writeBLE = null;

    if (typeof uniAdapter.writeBLECharacteristicValue === 'function') {
      writeBLE = uniAdapter.writeBLECharacteristicValue; // Named Export 模式
    } else if (uniAdapter.default && typeof uniAdapter.default.writeBLECharacteristicValue === 'function') {
      writeBLE = uniAdapter.default.writeBLECharacteristicValue; // Default Export 模式
    } else if (typeof window !== 'undefined' && window.uni && typeof window.uni.writeBLECharacteristicValue === 'function') {
      writeBLE = window.uni.writeBLECharacteristicValue; // 全域 Window 模式
    }

    if (!writeBLE) {
      console.error('找不到藍牙發送函數！目前的 uniAdapter:', uniAdapter);
      showToast({ type: 'fail', message: '蓝牙模组结构异常，请查看控制台' });
      return;
    }

    // 3. 執行寫入
    writeBLE({
      deviceId: bluetoothStore.connectedDevice.deviceId || 'web-device', 
      serviceId: '0000fff0-0000-1000-8000-00805f9b34fb',
      characteristicId: '0000fff2-0000-1000-8000-00805f9b34fb',
      value: recipePkg.buffer,
      success: () => {
        recipeShow.value = false
        showToast({ type: 'success', message: '已成功写入咖啡机专属定制' })
      },
      fail: (err: any) => {
        console.error('藍牙寫入失敗:', err)
        showToast({ type: 'fail', message: '写入失败，请确认蓝牙状态' })
      }
    })
    
  } catch (error) {
    console.error('打包或發送錯誤:', error)
    showToast({ type: 'fail', message: '发生未预期错误' })
  }
}

const downloadCloudRecipe = (recipe: Recipe) => {
  const isDuplicate = localRecipes.value.some(r => r.name === recipe.name.replace('【官方】', ''))
  if (isDuplicate) {
    showToast({ type: 'text', message: '该配方已存在于我的私藏中' })
    return
  }

  const downloaded: Recipe = {
    id: Date.now(),
    name: recipe.name.replace('【官方】', ''),
    powder: recipe.powder,
    ratio: recipe.ratio,
    temp: recipe.temp
  }

  localRecipes.value.unshift(downloaded)
  localStorage.setItem('bincoo_local_recipes', JSON.stringify(localRecipes.value))
  showToast({ type: 'success', message: '下载成功，已加入我的私藏' })
}

const confirmSaveRecipe = () => {
  const finalName = newRecipeName.value.trim() || `我的配方 ${new Date().toLocaleTimeString('zh-TW', {hour: '2-digit', minute:'2-digit'})}`
  
  const newRecipe: Recipe = {
    id: Date.now(), 
    name: finalName,
    powder: currentPowder.value,
    ratio: currentRatio.value,
    temp: currentTemp.value
  }

  localRecipes.value.unshift(newRecipe)
  localStorage.setItem('bincoo_local_recipes', JSON.stringify(localRecipes.value))
  showToast({ type: 'success', message: '已保存至我的私藏！' })
}

const runStateObj = computed(() => {
  const status = machineStatusStore.runStatus
  if (status === 1) {
    return { text: '研磨中', color: '#004097', icon: 'icon-yanyanjia', isJump: true }
  } else if (status === 2 || status === 3) {
    return { text: '冲煮中', color: '#004097', icon: 'icon-zhushui', isJump: true }
  }
  return { text: '设备就绪', color: '#999', icon: 'icon-kongxian', isJump: false }
})

const toHome = () => router.push('/')
const toQuick = () => router.push('/pages-coffeeb/grind/grind')
const toState = () => {
  if (runStateObj.value.isJump) {
    router.push('/pages-coffeeb/pulverizing/pulverizing')
  }
}

const colorConfirm = () => {
  showToast({ type: 'success', message: '颜色设置成功' })
}

onMounted(() => {
  const saved = localStorage.getItem('bincoo_local_recipes')
  if (saved) {
    try {
      localRecipes.value = JSON.parse(saved)
    } catch(e) {
      console.error('解析本地配方失败', e)
    }
  }
})
</script>
